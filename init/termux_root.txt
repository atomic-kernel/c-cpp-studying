cat /sys/kernel/debug/wakeup_sources

cmd power set-mode 1
cmd power set-fixed-performance-mode-enabled true
cmd power set-adaptive-power-saver-enabled false

stop mi_thermald
stop vendor.thermal-mediatek
stop thermald

cpus=$(nproc)
for i in $(seq 0 $((cpus - 1)))
do
  echo $i
  cd /sys/devices/system/cpu/cpu${i}/cpufreq
  max=$(cat scaling_available_frequencies | awk '{print $1}')
  echo performance > scaling_governor
  echo $max > scaling_max_freq
  echo $max > scaling_min_freq
  cd -
  cd /sys/devices/system/cpu/cpu${i}/cpuidle
  for i2 in state*
  do
    echo 1 > $i2/disable
  done
  cd -
done
