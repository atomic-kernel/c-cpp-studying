# 建议不配置apt镜像，直接跳过
# 选择软件包：如果是安装server版本，什么都不选；如果是安装桌面版本，建议只选择kde，其他软件包不需要

# /etc/default/grub 建议配置：
```
# 如果有显示器：
GRUB_CMDLINE_LINUX_DEFAULT="nouveau.modeset=1 amdgpu.modeset=1"

GRUB_DISABLE_OS_PROBER=true
```

# 多显卡配置主显卡（如屏蔽集显，使用独显输出）：
1. 查看显卡
lspci -nnk | grep -A3 -E "VGA|3D"
ls -al /dev/dri/by-path
ls -al /dev/dri/

2. 配置希望的主显卡 ~/.config/plasma-workspace/env/gpu.sh
```
export KWIN_DRM_DEVICES=/dev/dri/card0:/dev/dri/card1
```

# 多显示器配置主显示器：
1. cmdline 可选
fbcon=map:1
2. 在kde里面配置：Display configuration
3. 在kde里面配置：Color & Themes -> SDDM -> Apply Plasma Settings

apt --no-install-recommends install fcitx5-pinyin kde-config-fcitx5

# 精简
## 打印服务：
apt autopurge cups-daemon cups
## 磁盘健康：
apt autopurge smartmontools
## flatapk沙盒服务：
systemctl --user stop xdg-document-portal
systemctl --user mask xdg-document-portal
## systemd-journald:
systemctl stop systemd-journald.service systemd-journald.socket systemd-journald-dev-log.socket systemd-journal-flush.service
systemctl mask systemd-journald.service systemd-journald.socket systemd-journald-dev-log.socket systemd-journal-flush.service
## 文件共享：
apt autopurge kdeconnect
## gmenudbusmenuproxy
systemctl --user stop plasma-gmenudbusmenuproxy.service
systemctl --user mask plasma-gmenudbusmenuproxy.service
## xdg-permission-store: flatapk沙盒权限相关
systemctl --user stop xdg-permission-store
systemctl --user mask xdg-permission-store
## kaccess: 快捷键
## flatapk沙盒相关
systemctl --user mask xdg-desktop-portal.service
systemctl --user stop xdg-desktop-portal.service
## kactivitymanagerd: 禁用会黑屏
systemctl --user mask plasma-kactivitymanagerd.service
systemctl --user stop plasma-kactivitymanagerd.service
## packagekit: 软件包自动更新相关
apt autopurge packagekit plasma-discover appstream
